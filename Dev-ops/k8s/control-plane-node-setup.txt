Kubernetes setup:
======================================

Take 3 Amazon Machine Images : Ubuntu Server 24.04 LTS (HVM)
Instance Type : t3.small
Storage : 15 GB each.

control-plane setup:
======================================
1. change host name:
	sudo hostnamectl set-hostname control-plane

2. logout and login again.

3. 
create a file script.sh	
	touch script.sh

give execution permission to the script:
	chmod u+x script.sh

add following contents to it and execute the script.

sudo swapoff -a
sleep 2
echo "net.ipv4.ip_forward = 1" | sudo tee /etc/sysctl.d/k8s.conf
sleep 2
sudo sysctl --system
sleep 2
sysctl net.ipv4.ip_forward
sleep 2
sudo apt update && sudo apt upgrade -y
sleep 2
sudo apt-get install containerd -y
sleep 2
ctr --version
sleep 2
sudo mkdir -p /opt/cni/bin
CNI_VERSION=$(curl -sSL "https://api.github.com/repos/containernetworking/plugins/releases/latest" | jq -r '.tag_name')
wget -q "https://github.com/containernetworking/plugins/releases/download/${CNI_VERSION}/cni-plugins-linux-amd64-${CNI_VERSION}.tgz"
sudo tar Cxzf /opt/cni/bin "cni-plugins-linux-amd64-${CNI_VERSION}.tgz"
sleep 2
sudo mkdir /etc/containerd
sudo containerd config default | sudo tee /etc/containerd/config.toml
sleep 2
sudo sed -i -e 's/SystemdCgroup = false/SystemdCgroup = true/g' /etc/containerd/config.toml
sudo systemctl restart containerd
sleep 2
sudo systemctl enable containerd
sleep 2
sudo apt-get update
sleep 2
sudo apt-get install -y apt-transport-https ca-certificates curl gpg
sleep 2
sudo apt-get update
sleep 2
curl -fsSL https://pkgs.k8s.io/core:/stable:/v1.35/deb/Release.key | sudo gpg --dearmor -o /etc/apt/keyrings/kubernetes-apt-keyring.gpg
sleep 2
echo 'deb [signed-by=/etc/apt/keyrings/kubernetes-apt-keyring.gpg] https://pkgs.k8s.io/core:/stable:/v1.35/deb/ /' | sudo tee /etc/apt/sources.list.d/kubernetes.list
sudo apt-get update
sleep 2
sudo apt-get install -y kubelet kubeadm kubectl
sleep 2
sudo apt-mark hold kubelet kubeadm kubectl
sleep 2
sudo kubeadm init --pod-network-cidr=192.168.0.0/16
sleep 2
mkdir -p $HOME/.kube
sudo cp -i /etc/kubernetes/admin.conf $HOME/.kube/config
sudo chown $(id -u):$(id -g) $HOME/.kube/config
sleep 2
kubectl apply -f https://raw.githubusercontent.com/projectcalico/calico/v3.31.0/manifests/calico.yaml
sleep 2
echo "*------- Control Plane setup Sucess -------*"
